use jl;
create table logs (
	uuid char(36) not null primary key,
	userid varchar(128),
	username varchar(128),
	created_date bigint(20),
	message varchar(2048),
	input_text varchar(1024),
	code_text varchar(32768),
	output_text varchar(1024),
	watch_text varchar(1024),
	count_click_run int,
	count_click_step int,
	count_click_stop int
);
